An utility for managing user sessions

Demo:

  1.  LoadIBC1Class('UserPassport', 'data.user');
  2.  $up=new UserPassport('usertest');
  3.  try{
  4.      $up->Login('webmaster','mypwd');
  5.      echo "login!\n";
  6.      echo "welcome ".$up->GetUID()."\n";
  7.  } catch(Exception $ex) {
  8.      echo $ex->getMessage()."\n";
  9.  }
  10.  $up->Logout();
  11.  if(!$up->IsOnline()) echo "logout!\n";
